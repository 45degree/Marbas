layout(location = 1)in vec3 aNormal;
layout(location = 2)in vec2 aTexCoord;
layout(location = 3)in vec2 aTexID;

layout(location = 0)out vec2 ourTex;
layout(location = 1)out vec2 texId;

layout(std140, binding = 0)uniform Matrices {
    mat4 model;
    mat4 view;
    mat4 projection;
};

void main(){
    gl_Position = projection * view * model * vec4(aPos, 1.0);
    ourTex = aTexCoord;
    texId = aTexID;
}

layout(location = 0)in vec2 ourTex;
layout(location = 1)in vec2 texId;

layout(location = 0)out vec4 FragColor;

layout(binding = 0)uniform sampler2D ourTextures[32];

void main(){

    int id = int(texId[0]);
    if(id != - 1){
        vec4 texColor = texture(ourTextures[id], ourTex);

        if(texColor . a < 0.1)
            discard;

        FragColor = texColor;
    }
}
